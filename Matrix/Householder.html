Householder Bidiagonalization algorithm. MC, Golub, pg 252, Algorithm 5.4.2 Returns the matrices U_B and V_B such that: U_B^T * A * V_B = B, where B is upper bidiagonal. 

    # File lib/euclid/matrix/householder.rb, line 37
37:     def Householder.bidiag(mat)
38:       a = mat.clone
39:       m = a.row_size
40:       n = a.column_size
41:       ub = Matrix.I(m)
42:       vb = Matrix.I(n)
43:       n.times{|j|
44:         v, beta = a[j..m-1,j].house
45:         a[j..m-1, j..n-1] = (Matrix.I(m-j) - beta * (v * v.t)) * a[j..m-1, j..n-1]
46:         a[j+1..m-1, j] = v[1..(m-j-1)]
47:         ub *= bidiagUV(a[j+1..m-1,j], m, beta) #Ub = U_1 * U_2 * ... * U_n
48:         if j < n - 2
49:           v, beta = (a[j, j+1..n-1]).house
50:           a[j..m-1, j+1..n-1] = a[j..m-1, j+1..n-1] * (Matrix.I(n-j-1) - beta * (v * v.t))
51:           a[j, j+2..n-1] = v[1..n-j-2]
52:           vb  *= bidiagUV(a[j, j+2..n-1], n, beta) #Vb = V_1 * U_2 * ... * V_n-2
53:         end  }
54:       return ub, vb
55:     end

60:       h = mat.clone
61:       n = h.row_size
62:       u0 = Matrix.I(n)
63:       for k in (0...n - 2)
64:         v, beta = h[k+1..n-1, k].house #the householder matrice part
65:         houseV = Matrix.I(n-k-1) - beta * (v * v.t)
66:         u0 *= Matrix.diag(Matrix.I(k+1), houseV)
67:         h[k+1..n-1, k..n-1] = houseV * h[k+1..n-1, k..n-1]

69:       end
